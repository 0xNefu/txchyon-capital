// Helper function for automatic readTime calculation
function calculateReadTime(content, wordsPerMinute = 130) {
  if (!content) return 10;
  
  const plainText = content
    .replace(/---[\s\S]*?---/, '')
    .replace(/```[\s\S]*?```/g, '')
    .replace(/<[^>]*>/g, '')
    .replace(/#{1,6}\s*/g, '')
    .replace(/[\[\]\(\)]/g, ' ')
    .replace(/\s+/g, ' ')
    .trim();
  
  const wordCount = plainText.split(/\s+/).length;
  const minutes = Math.ceil(wordCount / wordsPerMinute);
  
  return Math.max(1, Math.min(minutes, 60));
}

function getPostReadTime(post) {
  if (post.data.readTime) return post.data.readTime;
  if (post.body) return calculateReadTime(post.body);
  
  // Estimates for all 11 pillars
  const estimates = {
    'getting-started': 8,
    'security-privacy': 10,
    'trading-investing': 12,
    'defi-yield': 15,
    'airdrop-farming': 12,
    'infrastructure-tech': 14,
    'tools-automation': 13,
    'research-analysis': 15,
    'regulatory-tax': 12,
    'portfolio-management': 12,
    'prediction-markets': 14
  };
  
  return estimates[post.data.pillar] || 10;
}

// Get all posts and add automatic readTime
const allPostsRaw = await getCollection('posts', ({ data }) => !data.draft);

const allPosts = allPostsRaw.map(post => ({
  ...post,
  data: {
    ...post.data,
    readTime: getPostReadTime(post)
  }
}));

// Sort by date, newest first
const sortedPosts = allPosts.sort(
  (a, b) => new Date(b.data.date).getTime() - new Date(a.data.date).getTime()
);

// Featured post (most recent - FIRST post)
const featuredPost = sortedPosts[0];

// LATEST POSTS: Skip the first one (featured) and take next 6
const latestPosts = sortedPosts.slice(1, 7);

// Get posts by all 11 pillars
const pillarPosts = {};
Object.keys(CATEGORY_HIERARCHY).forEach(pillar => {
  pillarPosts[pillar] = sortedPosts
    .filter(p => p.data.pillar === pillar && p.slug !== featuredPost?.slug)
    .slice(0, 3);
});

// FIXED: Get "getting-started" posts for Quick Wins
const gettingStartedPosts = sortedPosts
  .filter(p => p.data.pillar === 'getting-started' && p.slug !== featuredPost?.slug);

// If we have enough getting-started posts, use them
let quickWins;
if (gettingStartedPosts.length >= 4) {
  quickWins = gettingStartedPosts.slice(0, 4);
} else {
  // Fallback: Get posts from any pillar, excluding featured
  const otherPosts = sortedPosts
    .filter(p => p.slug !== featuredPost?.slug)
    .slice(0, 4 - gettingStartedPosts.length);
  
  quickWins = [...gettingStartedPosts, ...otherPosts].slice(0, 4);
}

// If still no posts (edge case), use latest posts
if (!quickWins || quickWins.length === 0) {
  quickWins = sortedPosts.slice(0, 4);
}

// Realistic metrics
const realisticStats = {
  guidesPublished: allPosts.length.toString(),
  activeReaders: "1,200+",
  protocolsCovered: "45+",
  monthsActive: "1"
};
